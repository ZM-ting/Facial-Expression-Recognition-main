# 员工考勤管理系统

基于 Python 开发的轻量级考勤管理系统，支持考勤数据录入、打卡状态判定（迟到/早退/缺勤）、工时统计、报表导出等核心功能；可选接入 SQLite/MySQL 数据库实现数据持久化存储，适配单人/小团队考勤管理场景。

# 项目介绍

## 核心功能

- 📊 考勤数据管理：支持 Excel/CSV 导入、数据库录入两种方式
- ⏰ 打卡状态自动判定：自定义上下班时间，自动识别迟到、早退、缺勤、正常打卡
- 📈 工时统计：按员工/日期/部门维度统计考勤数据，生成可视化报表（可选）
- 📤 结果导出：支持将考勤统计结果导出为 Excel/CSV 格式
- 💾 数据持久化：可选 SQLite（零配置）/MySQL（高可用）数据库存储，避免文件丢失

# 快速使用

## 方式 1：无数据库（轻量版，Excel 读写）

1. 准备考勤原始数据：
   将员工打卡数据按模板填写到 `data/raw_attendance.xlsx`（模板见 `data/template.xlsx`），字段规范：

   | 员工编号 | 员工姓名 | 打卡日期   | 上班打卡时间 | 下班打卡时间 |
   | -------- | -------- | ---------- | ------------ | ------------ |
   | 001      | 张三     | 2024-05-20 | 09:05        | 18:00        |

2. 运行核心脚本：
   python src/attendance_analysis.py

3. 查看结果：
   考勤统计结果会生成在 `output/attendance_report_20240520.xlsx`，控制台会输出关键统计信息（如迟到人数、总工时等）。

## 方式 2：接入数据库（推荐，持久化存储）

### 方案 A：SQLite（零配置，默认）

1. 初始化数据库（自动创建表结构）：
   python src/db_init.py

2. 导入数据到数据库：
   python src/import_to_db.py --file data/raw_attendance.xlsx

3. 数据库查询与统计：
   python src/db_attendance_analysis.py

### 方案 B：MySQL（需提前配置）

1. 修改 `config/db_config.py` 中的 MySQL 连接信息：
   MYSQL_CONFIG = {
   "host": "localhost",
   "user": "root",
   "password": "你的MySQL密码",
   "database": "attendance_db"
   }

2. 手动在 MySQL 中创建数据库：
   CREATE DATABASE IF NOT EXISTS attendance_db DEFAULT CHARSET utf8mb4;

3. 重复「方案 A」的步骤 1-3 完成数据导入和统计。

def load_excel_data(file_path):
"""加载Excel考勤数据"""
try:
df = pd.read_excel(file_path) # 数据清洗：去除空行、规范字段名
df = df.dropna(how="all")
df.columns = df.columns.str.strip() # 去除字段名空格
required_cols = ["员工编号", "员工姓名", "打卡日期", "上班打卡时间", "下班打卡时间"]
for col in required_cols:
if col not in df.columns:
raise ValueError(f"缺少必要字段：{col}")
return df
except Exception as e:
print(f"加载Excel数据失败：{e}")
return None

def import_to_sqlite(df):
"""导入数据到SQLite"""
try:
conn = sqlite3.connect(SQLITE_CONFIG["db_path"])
cursor = conn.cursor()

        # 批量插入数据
        for _, row in df.iterrows():
            # 格式转换
            check_date = pd.to_datetime(row["打卡日期"]).date()
            check_in = row["上班打卡时间"] if pd.notna(row["上班打卡时间"]) else None
            check_out = row["下班打卡时间"] if pd.notna(row["下班打卡时间"]) else None

            # 插入SQL
            insert_sql = """
            INSERT INTO employee_attendance
            (employee_id, employee_name, check_date, check_in_time, check_out_time)
            VALUES (?, ?, ?, ?, ?)
            """
            cursor.execute(insert_sql, (
                row["员工编号"], row["员工姓名"], check_date, check_in, check_out
            ))

        conn.commit()
        print(f"成功导入 {len(df)} 条数据到SQLite！")
    except Exception as e:
        print(f"导入SQLite失败：{e}")
        conn.rollback()
    finally:
        if conn:
            conn.close()

# 无数据库版考勤分析脚本

import pandas as pd
from datetime import datetime, time
import os
from src.attendance_rules import (
WORK_START_TIME, WORK_END_TIME,
LATE_THRESHOLD, EARLY_LEAVE_THRESHOLD
)

# 转换时间字符串为time对象

def str_to_time(time_str):
"""将HH:MM格式的字符串转换为time对象"""
if pd.isna(time_str) or time_str == "":
return None
return datetime.strptime(str(time_str), "%H:%M").time()

def calculate_work_hours(check_in, check_out):
"""计算工时（小时）"""
if not check_in or not check_out:
return 0.0 # 转换为datetime计算差值
today = datetime.now().date()
in_dt = datetime.combine(today, check_in)
out_dt = datetime.combine(today, check_out)
hours = (out_dt - in_dt).total_seconds() / 3600
return round(hours, 2)

def judge_attendance_status(check_in, check_out):
"""判定考勤状态"""
work_start = str_to_time(WORK_START_TIME)
work_end = str_to_time(WORK_END_TIME)

    # 缺勤：无上班打卡
    if not check_in:
        return "absent"

    # 迟到判定
    late_time = datetime.combine(datetime.now().date(), work_start)
    late_time = late_time.replace(minute=late_time.minute + LATE_THRESHOLD)
    late_time = late_time.time()

    # 早退判定
    early_time = datetime.combine(datetime.now().date(), work_end)
    early_time = early_time.replace(minute=early_time.minute - EARLY_LEAVE_THRESHOLD)
    early_time = early_time.time()

    status = "normal"
    if check_in > late_time:
        status = "late"
    if check_out and check_out < early_time:
        status = "early_leave" if status == "normal" else "late_early_leave"

    return status

def analyze_attendance(file_path, output_dir="output"):
"""核心考勤分析逻辑""" # 创建输出目录
if not os.path.exists(output_dir):
os.makedirs(output_dir)

    # 加载数据
    df = pd.read_excel(file_path)
    df.columns = df.columns.str.strip()

    # 数据处理
    df["上班打卡时间"] = df["上班打卡时间"].apply(str_to_time)
    df["下班打卡时间"] = df["下班打卡时间"].apply(str_to_time)

    # 判定状态、计算工时
    df["考勤状态"] = df.apply(
        lambda x: judge_attendance_status(x["上班打卡时间"], x["下班打卡时间"]),
        axis=1
    )
    df["工时(小时)"] = df.apply(
        lambda x: calculate_work_hours(x["上班打卡时间"], x["下班打卡时间"]),
        axis=1
    )

    # 统计汇总
    total_employees = df["员工编号"].nunique()
    late_count = len(df[df["考勤状态"].str.contains("late")])
    early_leave_count = len(df[df["考勤状态"].str.contains("early_leave")])
    absent_count = len(df[df["考勤状态"] == "absent"])
    total_work_hours = df["工时(小时)"].sum()

    # 打印统计结果
    print("="*50)
    print("考勤统计结果")
    print("="*50)
    print(f"总员工数：{total_employees}")
    print(f"迟到人数：{late_count}")
    print(f"早退人数：{early_leave_count}")
    print(f"缺勤人数：{absent_count}")
    print(f"总工时：{total_work_hours} 小时")
    print("="*50)

    # 导出报表
    output_file = os.path.join(
        output_dir,
        f"attendance_report_{datetime.now().strftime('%Y%m%d')}.xlsx"
    )
    df.to_excel(output_file, index=False)
    print(f"考勤报表已导出至：{output_file}")

    return df
